CREATE TABLE avioane(
ID_AVION NUMBER(3) PRIMARY KEY,
MODEL VARCHAR2(20),
AUTONOMIE NUMBER(5),
NR_LOCURI NUMBER(3)
);

CREATE TABLE aeroporturi(
ID_AEROPORT VARCHAR2(3) PRIMARY KEY,
NUME VARCHAR2(30),
PROCESARI_MAX_ZILNIC NUMBER(2)
);

CREATE TABLE rute(
ID_RUTA NUMBER(3) PRIMARY KEY,
PLECARE VARCHAR2(20),
SOSIRE VARCHAR2(20),
DISTANTA NUMBER(5),
CONSTRAINT PLECARE_FK FOREIGN KEY(PLECARE) REFERENCES AEROPORTURI(ID_AEROPORT),
CONSTRAINT SOSIRE_FK FOREIGN KEY(SOSIRE) REFERENCES AEROPORTURI(ID_AEROPORT)
);

CREATE TABLE zboruri(
ID_ZBOR NUMBER(3) PRIMARY KEY,
ID_RUTA NUMBER(3),
ID_AVION NUMBER(3),
ZIUA DATE DEFAULT SYSDATE,
CONSTRAINT ID_RUTA_FK FOREIGN KEY(ID_RUTA) REFERENCES RUTE(ID_RUTA),
CONSTRAINT ID_AVION_FK FOREIGN KEY(ID_AVION) REFERENCES AVIOANE(ID_AVION)
);

CREATE TABLE GRAD_DE_OCUPARE(
DATA DATE PRIMARY KEY);
